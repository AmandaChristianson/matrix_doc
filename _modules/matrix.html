"""
Matrix object with operations.

The matrix is an object which stores the elements of the matrix in a nested list. The 
library comes with both methods as part of the class and external functions useful for 
matrices.

The methods use Python's standard notation for instance constructing, 
addition, subtraction and multiplication. The standard multiplication is implemented 
to support both matrix- and scalar-matrix multiplication, depending on input.
Additional functionality include element-wise multiplication and division 
as well as computing the trace and transpose of a matrix. Finally, 
the function identity(n) constructs an identity matrix with dimension n*n.
"""

[docs]    def __init__(self, data):
        """
        | Initates a matrix instance.
        | Parameters: data=nested list with int or float elements
        | Input format: [[],[],...,[]] where the comma-seperated nested lists holds the elements in each row of the matrix.
        """
        self._check_input(data) # Raises exceptions if data consists of incorrect types
        
        # Assignment
        self._data = data
        self._rows = len(data)
        self._columns = max(1, len(data[0]))
        self._dimensions = (self._rows, self._columns)


    def _check_input(self, data): # Raises exceptions if data consists of incorrect types
        if not isinstance(data, list): # Checks that the matrix is a list
            raise TypeError("The matrix must be a (nested) list.")

        for i in range(len(data)): # Checks that each row is a list
            if not isinstance(data[i], list):
                raise TypeError("Each row must be a list.")

        for i in range(len(data)): # Checks that there is an equal number of elements in each row
            for j in range(len(data[0])-1):
                if len(data[i]) != len(data[0]):
                    raise Exception("Each row must have an equal number of elements.")

        for i in range(len(data)): # Checks that elements are ints or floats
            for j in range(len(data[0])):
                if not (isinstance(data[i][j], int) or isinstance(data[i][j], float)):
                    raise TypeError("The elements in the matrix must be of type int or float.")


[docs]    def __str__(self):
        """
        | String representation of the matrix.
        | Parameters: self=matrix
        | Returns: string
        | Time complexity: O(m*n), m*n is the dimension of the matrix.
        """
        m_str = "["
        for i in range(self._rows-1):
            m_str += (str(self._data[i]) + "\n")
        m_str += str(self._data[self._rows-1]) + "]" # Last row in matrix

        return m_str

        """

        | Called by writing "self + B".
        | Parameters: self=matrix, B=matrix
        | Returns: nested list
        | Time complexity: O(m*n), m*n is the dimension of the matrices.
        """
        if self._dimensions == B._dimensions:
            rows,columns = self._dimensions
            res = _zeros(rows, columns)
            for i in range(rows):
                for j in range(columns):
                    res._data[i][j] = self._data[i][j] + B._data[i][j]
        else:
            raise Exception("The matrices must have the same dimensions.")

        return res._data

[docs]    def __mul__(self,x):
        """
        | Matrix or matrix-scalar multiplication.
        | Called by writing "self * x" where self is a matrix.
        | Returns the result of scalar-matrix multiplication if x is a scalar (float or int).
        | Returns the result of matrix multiplication self * x if x is a matrix.
        | 
        | Requirement for matrix multiplication: self._columns == x._rows.
        | 
        | Parameters: self = matrix, x = matrix, int or float
        | Returns: nested list
        | 
        | Time complexity of scalar multiplication: O(m*n) where m*n is the dimension of the matrix.
        | Time complexity of matrix multiplication: O(m*n*k) where m is the number of rows in self, 
        | n number of columns in self (=number of rows in x) and k number of columns in x.

        """
        if isinstance(x, matrix): # Matrix multiplication
            return _matrix_mul(self,x)
        elif isinstance(x, int) or isinstance(x, float): # Matrix-scalar multiplication
            return _scalar_mul(self,x)
        else:
            raise TypeError("Only integers, floats and matrices can be multiplied with a matrix.")

        """
        | Transpose of the matrix.
        | Parameters: self=matrix
        | Returns: nested list
        | Time complexity: O(m*n) where (m*n) is the dimension on the matrix.
        """
        rows,columns = self._rows, self._columns
        T = _zeros(columns,rows)
        for i in range(rows):
            for j in range(columns):
                T._data[j][i] = self._data[i][j]
        return T._data



[docs]    def trace(self):
        """
        | Trace of a square matrix.
        | Parameters: self=matrix
        | Returns: int or float
        | Time complexity: O(m), where m is the number of rows (= number of columns) in the matrix.
        """
        if self._rows == self._columns:
            res = 0
            for i in range(self._rows):
                res += self._data[i][i] # Adds the diagonal elements
        else:
            raise Exception("Trace is only defined for square matrices.")
        return res

def _zeros(rows,columns):
    """
    | Zero matrix with dimension (rows,columns).
    | Parameters: rows=int, columns=int
    | Returns: matrix
    | Time complexity: O(m*n) where (m*n) is the dimension on the matrix.
    """
    if (not isinstance(rows, int)) or rows <= 0:
        raise TypeError("rows must be a non-negative integer")
    if (not isinstance(columns, int)) or columns <= 0:
        raise TypeError("columns must be a non-negative integer")

    M = [[0]*columns for _ in range(rows)]
    return matrix(M)


[docs]def identity(n):
    """
    | Identity matrix with dimension nxn, where n is a non-negative integer.
    | Parameters: n=int
    | Returns: matrix
    | Time complexity: O(n).
    """
    if (not isinstance(n, int)) or n <= 0:
        raise TypeError("n must be a non-negative integer")

    I = [[0]*n for _ in range(n)] # Initiates a nested zero list
    for i in range(n): # Changes the diagonal elements to ones
        I[i][i] = 1

    return matrix(I)

[docs]def multiply(A,B):
    """
    | Element-wise multiplication of two matrices of equal dimensions.
    | Parameters: A=matrix, B=matrix
    | Returns: nested list
    | Time complexity: O(m*n) were m*n is the dimension of the matrices.
    """
    if A._dimensions == B._dimensions:
        rows,columns = A._dimensions
        res = _zeros(rows, columns)
        for i in range(rows):
            for j in range(columns):
                res._data[i][j] = A._data[i][j] * B._data[i][j]
    else:
        raise Exception("The matrices must have the same dimensions")
    
    return res._data

def _scalar_mul(A,k):
    """
    | Returns the scalar multiplication A*k for matrix A and scalar k (float or int).
    | Parameters: A=matrix, k=int or float
    | Returns: nested list
    | Time complexity: O(m*n), where m*n is the dimension of A.
    """
    rows,columns = A._dimensions
    res = _zeros(rows,columns)
    for i in range(rows):
        for j in range(columns):
            res._data[i][j] = A._data[i][j] * k
    return res._data

def _matrix_mul(A,B):
    """
    | Returns the matrix multiplication A*B for matrices A and B.
    | Requires that A._columns = B._rows
    | Parameters: A=matrix, B=matrix
    | Returns: nested list
    | Time complexity: O(m*n*k) where m is the number of rows in A, n number of columns in A (=number of rows in B) and k number of columns in B.
    """
    if A._columns == B._rows:
        res = _zeros(A._rows, B._columns)
        for i in range(A._rows):
            for j in range(B._columns):
                for k in range(B._rows):
                    res._data[i][j] += A._data[i][k] * B._data[k][j]
    else:
        raise Exception("Non-compatible dimensions. The number of columns in A must be the same as the number of rows in B.") 

    return res._data
